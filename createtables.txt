CREATE TABLE TRIP_HISTORY ( 
	S_no varchar (8) NOT NULL,
	Customer_id varchar (12),
	Driver_id varchar (12),
	From_location varchar (15) NOT NULL,
	To_location varchar (15) NOT NULL,
	From_time time NOT NULL,
	To_time time,
	trip_date date,
	Amount varchar (8),
	PRIMARY KEY (Customer_id,Driver_id,From_time),
	CONSTRAINT trip_refer_customer_id FOREIGN KEY (customer_id) REFERENCES CUSTOMER (user_id),
	CONSTRAINT trip_refer_driver_id FOREIGN KEY (Driver_id) REFERENCES DRIVER (user_id)

);
